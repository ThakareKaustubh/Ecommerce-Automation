# test_user_registration.py

import pytest
from pages.login_page import LoginPage
from pages.registration_page import RegistrationPage

def test_user_registration(setup, base_url, test_data):
    driver = setup
    user_data = test_data["registration_data"]
    driver.get(base_url)

    login_page = LoginPage(driver)
    registration_page = RegistrationPage(driver)

    assert login_page.check_loginpage_load(), "Login page did not load"
    login_page.goto_login_signup_page()
    assert login_page.verify_new_user_signup(), "New user signup section not visible"

    login_page.complete_registration_step_1(user_data["username"], user_data["email"])
    assert login_page.verify_registration_submit_text(), "Registration submit failed"

    registration_page.fill_user_registration_form(user_data)
    registration_page.submit_form()
    assert registration_page.verify_account_created(), "Account creation failed"

    registration_page.continue_post_reg()
    assert user_data["username"] in registration_page.check_logged_in_user(), "User not logged in"

    registration_page.delete_created_account()
    assert registration_page.delete_account_confirmation(), "Account deletion failed"
